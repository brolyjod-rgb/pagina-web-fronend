// recipes.js import { supabase } from './supabaseClient.js'; // importa tu cliente Supabase document.addEventListener("DOMContentLoaded", () => { const container = document.getElementById("recipesContainer"); if (!container) return; // ===== CARGAR RECETAS DESDE SUPABASE ===== async function loadRecipes() { try { const { data: recipes, error } = await supabase .from('recipes') // nombre de la tabla en Supabase .select('*') .order('createdAt', { ascending: false }); if (error) throw error; if (!recipes || recipes.length === 0) { container.innerHTML = `

No hay recetas guardadas aún. Agrega una nueva.

`; return; } // Mostrar cada receta en tarjeta recipes.forEach(recipe => { const card = document.createElement("div"); card.classList.add("recipe-card"); card.innerHTML = `

${recipe.title}

Categoría: ${recipe.category}

Tiempo: ${recipe.time}

Porciones: ${recipe.servings}

Calorías: ${recipe.calories}

Ingredientes:
${recipe.ingredients.join("
")}

Pasos:
${recipe.steps.join("
")}

${recipe.description}

`; // Click → abrir detalle card.addEventListener("click", () => { window.location.href = `recipe.html?id=${recipe.id}`; }); container.appendChild(card); }); } catch (error) { console.error(error); container.innerHTML = "

No se pudieron cargar las recetas ❌

"; } } loadRecipes(); });

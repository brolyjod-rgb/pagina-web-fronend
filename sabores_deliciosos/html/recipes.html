// === recipes.js === document.addEventListener("DOMContentLoaded", () => { const container = document.getElementById("recipesContainer"); if (!container) return; // Cargar recetas desde localStorage const recipes = JSON.parse(localStorage.getItem("recipes")) || []; // Mostrar mensaje si no hay recetas if (recipes.length === 0) { container.innerHTML = `

No hay recetas guardadas aún. Agrega una nueva.

`; return; } // Mostrar cada receta en tarjeta recipes.forEach(recipe => { const card = document.createElement("div"); card.classList.add("recipe-card"); card.innerHTML = `

${recipe.title}

Categoría: ${recipe.category}

Tiempo: ${recipe.time}

Porciones: ${recipe.servings}

Calorías: ${recipe.calories}

Ingredientes:
${recipe.ingredients.join("
")}

Pasos:
${recipe.steps.join("
")}

${recipe.description}

`; // Click → abrir detalle (opcional) card.addEventListener("click", () => { // Redirigir a recipe.html con query param window.location.href = `recipe.html?id=${recipe.id}`; }); container.appendChild(card); }); });
